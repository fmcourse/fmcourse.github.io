    \section{Введение}

	\begin{frame}{Почему Event-B}
    \begin{itemize}
		\item Сертификация по 4 уровню доверия к СЗИ требует 
		      разработать формальную модель управления доступом
			  и верифицировать ее с применением инструментальных
			  средств
		\item ГОСТ Р 59453.Х-2021 Защита информации. Формальная
		      модель управления доступом
		\item Event-B/Rodin - основные средства моделирования
			  и верификации формальных моделей управления доступом
	\end{itemize}
    \end{frame}

	\begin{frame}{Свойства Event-B/Rodin}
		\begin{itemize}
			\item формальный
			\item статически типизированный
			\item минималистичность
			\item математическая основа --- теория множеств
			\item наличие инструментальных средств для написания
			      моделей на Event-B и верификации (Rodin),
				  анимации (ProB) и др.
		\end{itemize}
	\end{frame}

	\begin{frame}{Дискретно-событийное моделирование}
		\begin{itemize}
			\item система обладает состоянием
			\item событие --- воздействие на систему,
					система мгновенно дает отклик
					и может изменить свое состояние
			\item события упорядочены
			\item система может включать программы,
				  аппаратуру, внешнюю среду
		\end{itemize}
	\end{frame}

	\begin{frame}{Моделирование --- это не программирование}
		Цель этой деятельности --- не написание
		алгоритма/программы на "еще одном языке",
		а получение правильных требований к программе/системе:

		\begin{enumerate}
			\item Точные, недвусмысленные формулировки
			\item Абстракция (отсутствие лишних деталей $\rightarrow$ проще)
			\item Доказана непротиворечивость
			\item Возможна поэтапная разработка модели (добавление
			      информации небольшими порциями с постоянным
				  доказательством корректности)
		\end{enumerate}
	\end{frame}

	\begin{frame}{Основные элементы формального метода Event-B}
		\begin{itemize}
			\item статические свойства системы --- `context`
			\item динамические свойства системы --- `machine`
			\item формальная модель --- множество контекстов и
			      последовательность машин
			\item контест может <<расширять>> другие контексты 
			\item машина может <<видеть>> несколько контекстов
			\item машина может <<уточнять>> другую машину
		\end{itemize}
	\end{frame}

	\section{Типы данных и выражения}

	\begin{frame}{Простейшие типы данных}
		\begin{itemize}
			\item \texttt{INT}
			\item \texttt{NAT}
			\item \texttt{NAT1}
			\item \texttt{BOOL}
			\item несущие множества
		\end{itemize}
	\end{frame}

	\begin{frame}{Выражения --- общие положения}
		\begin{itemize}
			\item статически типизированы
			\item могут быть неопределены (пример: делят на 0),
			      есть wd-условие
			\item логика первого порядка: предикаты
			      и выражения синтаксически различаются
				  (значением переменной не может быть предикат,
				   но может быть функция)
		\end{itemize}
	\end{frame}

	\begin{frame}[fragile]{Логические выражения}
		\begin{itemize}
			\item константы \verb|TRUE|, \verb|FALSE|
			\item логические связки \verb|&|, \verb|or|,
			      \verb|=>|, \verb|<=>|, \verb|not|
			\item логика короткая, при использовании
			      различных связок в одном выражении
			      скобки необходимы
			\item квантор всеобщности \verb|!|
			\item квантор существования \verb|#|
			\item сравнения \verb|=|, \verb|/=|, \verb|>|, ...
			\item принадлежность \verb|:|, \verb|/:|, \verb|<<|, ...
		\end{itemize}
	\end{frame}

	\begin{frame}[fragile]{Выражения с множествами}
		\begin{itemize}
			\item пустое множество \verb|{}|
			\item перечисление \verb|{1, 2, 3}|
			\item генератор \verb`{x * 2 | x : 1 .. 10}`
			\item целочисленные диапазоны (обе границы включаются) \verb|a .. b|
			\item теоретико-множественные операции \verb|\/|, \verb|/\|, \verb|\|
		\end{itemize}
	\end{frame}

	\begin{frame}[fragile]{Предопределенные функции и предикаты}
		\begin{itemize}
			\item предикат как выражение типа \texttt{BOOL} \verb|bool(predicate)|
			\item конечность множества \verb|finite(set)|
			\item размер множества \verb|card(set)|
			\item равенство объединению непересекающихся множеств
			      \verb|partition(set, set1, set2, ..., setN)|
		\end{itemize}
	\end{frame}

	\begin{frame}[fragile]{Пары и отношения}
		\begin{itemize}
			\item пара \verb`a |-> b`
			\item декартово произведение \verb|A ** B|
			\item множество подмножеств \verb|POW(A)|, \verb|POW1(A)|
			\item произвольное отношение \texttt{A <-> B}
			\item тотальное отношение \verb`A <<-> B`
			\item сюръективное отношение \verb`A <->> B`
			\item тотальное сюръективное отношение \verb`A <<->> B`
		\end{itemize}
	\end{frame}

	\begin{frame}[fragile]{Операции над отношениями}
		\begin{itemize}
			\item область определения (домен) \verb|dom(relation)|
			\item область значений \verb|ran(relation)|
			\item проекция \verb|relation[set]|
			\item оставить только поддомен \verb`set <| relation`
			\item убрать поддомен \verb`set <<| relation`
			\item оставить только значения \verb`relation |> set`
			\item убрать значения \verb`relation |>> set`
		\end{itemize}
	\end{frame}

	\begin{frame}[fragile]{Функции}
		Функция --- это детерминированное подмножество отношения.

		\begin{itemize}
			\item множество тотальных функций \verb|A -> B|
			\item множество частичных функций \verb|A +-> B|
			\item применение \verb|function(arg)|
		\end{itemize}
	\end{frame}

	\begin{frame}[fragile]{Действия}
		\begin{itemize}
			\item одиночное \verb|x := E|
			\item множественное \verb|x1, x2, ..., xN := E1, E2, ..., EN|
			\item замена пары в отношении \verb|x(F) := E|
			\item неявное одиночное  \verb`x :| before-after-predicate with x and x'`
			\item неявное множественное
			      \verb`x1, ..., xN :| b-a-p with x1, ..., xN, x1', ..., xN'`
			\item неявный выбор из непустого множества \verb`x :: set`
		\end{itemize}
	\end{frame}

	\section{Семантика}

    \begin{frame}{Семантика контекстов}
        \begin{enumerate}
            \item Несущие множества и константы --- символы
            \item Интерпретация символов --- это сопоставление им
                    математических значений, при котором все
                    аксиомы истинны
            \item Формула является следствием из аксиом, если
                    она истинна для всех интерпретаций
            \item К условию верификации нужно относиться как
                    к следствию из аксиом 
        \end{enumerate}
    \end{frame}

	\begin{frame}{Функционирование машины}
		\begin{enumerate}
			\item состояние машины --- значения всех переменных
			\item событие --- переход из одного состояния в другое
			\item событие может иметь параметры
			\item переход возможен, только если истинны
			      все охранные условия
			\item в этом случае выполняются все действия
			\item действия одного события выполняются одновременно
			\item начальное состояние --- результат события \texttt{INITIALISATION}
		\end{enumerate}
	\end{frame}

	\begin{frame}{Корректно определенная машина/контекст}
		\begin{enumerate}
			\item \texttt{WD}: нет неопределенных выражений
			\item \texttt{INV}: не нарушены инварианты
			\item \texttt{THM}: не нарушены теоремы
			\item \texttt{FIS}: неявные действия разрешимы
			\item \texttt{GRD}, \texttt{SIM}, \texttt{EQL}:
			      уточнение корректно
		\end{enumerate}

		Rodin генерирует условия верификации для формального
		доказательства, что машина корректно определена.
	\end{frame}

    \section{Уточнение машины}

    \begin{frame}{Цель уточнения}
        \begin{itemize}

        \item
        Уточнение --- это отношение между двумя машинами,
        абстрактной и конкретной, при котором все свойства
        абстрактной машины справедливы для конкретной.

        \item
        Вертикальное уточнение (уточнение данных) --- конкретная
        машина понижает уровень абстракции (например, заменяет
        абстрактные значения на значения, близкие к реализации)

        \item
        Горизонтальное уточнение (уточнение суперпозицией) ---
        конкретная машина оставляет уровень абстракции тем же,
        но добавляет новые элементы машины (переменные, события). 
        \end{itemize}
    \end{frame}

    \begin{frame}{Состояния машин}
        \begin{itemize}
            \item В общем случае, состояния машин не связаны
            и могут отличаться (вертикальное уточнение)
            \item Если конкретная переменная использует то же
            имя переменной, что есть в абстрактной машине, у этих
            переменных одинаковое значение
            \item Склеивающий инвариант --- инвариант, описывающий
            соответствие между состояниями двух машин: состоянию
            конкретной машины соответствуют все состояния абстрактной
            машины, при которых склеивающий инвариант истинен
        \end{itemize}
    \end{frame}

    \begin{frame}{События машин}
        \begin{itemize}
            \item Каждое событие конкретной машины должно
                указать одно из событий абстрактной машины
                (если не указано, то используется skip, то есть
                 пустое действие)
            \item Параметры события конкретной машины
                могут отличаться от параметров абстрактной машины
            \item Для параметров абстрактной машины, отсутствующих
                в конкретной машине, нужно описать \empty{witness-условие}
                в событии конкретной машины
            \item Witness-условие --- это предикат от параметров события
                конкретной машины и параметра абстрактной машины,
                описывающий соответствие между ними
            \item Имя witness-условия должно совпадать с именем параметра
                абстрактной машины
        \end{itemize}
    \end{frame}

    \begin{frame}{Трассовое понимание уточнения}
        \begin{itemize}
        \item
        Конкретная трасса --- это цепочка состояний конкретной
        машины, между которыми находятся переходы конкретной
        машины такие, что все охранные условия и действия выполнены
        \item
        Абстрактная трасса --- аналогично для абстрактной машины
        \item
        Каждой конкретной трассе должна соответствовать
        хотя бы одна абстрактная трасса такая, что:
        \begin{enumerate}
        \item
        Каждому состоянию в конкретной трассе соответствует
        некоторое состояние в абстрактной трассе такое,
        что выполнен \emph{склеивающий инвариант}
        \item
        Каждому переходу в конкретной трассе соответствует
        некоторый переход в абстрактной машине
        \end{enumerate}
        \end{itemize}
    \end{frame}
    
    \begin{frame}{Трассовое понимание уточнения (2)}
        Иными словами, абстрактная машина описывает ограничения
        на безопасные трассы, а трассам конкретной машины должны
        соответствовать только безопасные трассы абстрактной машины.
    \end{frame}

    \begin{frame}{Условия верификации}
        Rodin генерирует условия верификации для формального
        доказательства этого свойства:
        \begin{enumerate}
        \item если охранное условие конкретного события истинно,
        то охранное условие абстрактного события тоже истинно
        \item для каждого конкретного перехода
        $C_1 \overset{e_C}{\rightarrow} C_2$ и каждого
        абстрактного перехода $A_1 \overset{e_A}{\rightarrow} A_2$
        для соответствующего события каждому $A_1$, склеенному с $C_1$,
        и $A_2$, склеенному с $C_2$, $A_2$ должен получаться
        в результате действия события $e_A$ из состояния $A_1$
        \end{enumerate}  
    \end{frame}
    
    \begin{frame}[fragile]{Синтаксис уточнения}
        \begin{itemize}
            \item
            \verb`machine M2 refines M1 ... end`
            задает, какая машина является абстрактной (\verb`M1`),
            а какая --- конкретной (\verb`M2`)
            \item
            \verb`event e2 refines e1 ... end`
            задает, на какое абстрактное событие (\verb`e1`)
            указывает конкретное событие (\verb`e2`)
            \item
            \verb`event ... any a b c with @p: predicate(p, a, b, c) ...`
            задает witness-условие для абстрактного параметра p,
            отсутствующего среди конкретных параметров (a, b, c)
        \end{itemize}
    \end{frame}

    \begin{frame}{События-расширения}
        Горизонтальное уточнение задается при помощи
          \emph{расширения событий}:
            \begin{enumerate}
            \item параметры только добавляются
            \item охранные условия только добавляются
            \item действия только добавляются
            \end{enumerate}

        В Rodin у таких событий написано \texttt{extended}
        и редактор показывает, но не позволяет редактировать,
        части абстрактного события
    \end{frame}
